import pandas as pd import matplotlib.pyplot as plt import seaborn as sns from pyscript import display # -------- Data -------- data = { 'Month': ['January', 'February', 'March', 'April'], 'TargetSales': [50000, 55000, 60000, 65000], 'ActualSales': [48000, 60000, 58000, 70000] } df = pd.DataFrame(data) # -------- Calculations -------- df['Performance'] = (df['ActualSales'] / df['TargetSales']) * 100 df['Growth'] = df['ActualSales'].pct_change() * 100 # -------- Display Table -------- display(df, target="table") # -------- Summary -------- overall = (df['ActualSales'].sum() / df['TargetSales'].sum()) * 100 best_month = df.loc[df['ActualSales'].idxmax()]['Month'] display(f"Overall Performance: {round(overall,2)}%", target="summary") display(f"Best Month: {best_month}", target="summary") # -------- Bar Chart -------- fig1, ax1 = plt.subplots() df.plot(x='Month', y=['TargetSales','ActualSales'], kind='bar', ax=ax1) ax1.set_title("Target vs Actual Sales") display(fig1, target="bar") # -------- Line Chart -------- fig2, ax2 = plt.subplots() sns.lineplot(x='Month', y='Performance', data=df, marker='o', ax=ax2) ax2.set_title("Performance (%)") display(fig2, target="line") # -------- Pie Chart -------- fig3, ax3 = plt.subplots() ax3.pie(df['ActualSales'], labels=df['Month'], autopct='%1.1f%%') ax3.set_title("Sales Contribution") display(fig3, target="pie") # -------- Growth Chart -------- fig4, ax4 = plt.subplots() sns.barplot(x='Month', y='Growth', data=df, ax=ax4) ax4.set_title("Month-to-Month Growth (%)") display(fig4, target="growth")
